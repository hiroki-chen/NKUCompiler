57 void compiler::Item_stmt_eif::generate_ir_helper(
58  compiler::ir::IRContext* const ir_context,
59  std::vector<compiler::ir::IR>& ir_list) const {
60  try {
61  ir_context->enter_scope();
62  // Get the id of current scope.
63  const uint32_t scope_uuid_cur =
64  (uint32_t)(ir_context->get_symbol_table()->get_top_scope_uuid());
65  const compiler::ir::BranchIR branch_ir =
66  condition->eval_cond(ir_context, ir_list);
67 
68  // The if_branch is taken. Evaluated at runtime.
69  if (branch_ir.first == compiler::ir::JMP) {
70  if_branch->generate_ir(ir_context, ir_list);
71  return;
72  }
73  if (branch_ir.second == compiler::ir::JMP) {
74  else_branch->generate_ir(ir_context, ir_list);
75  return;
76  }
77 
78  // If branches cannot be evaluated at runtime, we should create IRs for each
79  // branch.
80  ir_list.emplace_back(branch_ir.second,
81  ".LBB" + std::to_string(scope_uuid_cur) + "_ELSE");
82  ir_list.emplace_back(ir::op_type::JMP,
83  ".LBB" + std::to_string(scope_uuid_cur) + "_IF");
84 
85  // Generate a label for if branch.
86  ir_list.emplace_back(ir::op_type::LBL,
87  ".LBB" + std::to_string(scope_uuid_cur) + "_IF:");
88 
89  std::vector<compiler::ir::IR> ir_if, ir_else;
90  compiler::ir::IRContext ir_context_if(*ir_context);
91  compiler::ir::IRContext ir_context_else(*ir_context);
92 
93  ir_context_if.enter_scope();
94  if_branch->generate_ir(&ir_context_if, ir_if);
95  ir_context_if.leave_scope();
96 
97  ir_context_else.get_symbol_table()->set_available_id(
98  ir_context_if.get_symbol_table()->get_available_id());
99  ir_context_else.enter_scope();
100  else_branch->generate_ir(&ir_context_else, ir_else);
101  ir_context_else.leave_scope();
102 
103  ir_context->get_symbol_table()->set_available_id(
104  ir_context_else.get_symbol_table()->get_available_id());
105 
106  std::vector<compiler::ir::IR> ir_end;
107  ir_end.emplace_back(compiler::ir::op_type::LBL,
108  ".LBB" + std::to_string(scope_uuid_cur) + "_END_IF:");
109 
110  const auto symbol_table_if =
111  ir_context_if.get_symbol_table()->get_symbol_table();
112 
113  for (uint32_t i = 0; i < symbol_table_if.size(); i++) {
114  const auto symbol_table =
115  ir_context_else.get_symbol_table()->get_symbol_table()[i];
116  for (auto symbol_pair : *(symbol_table_if[i]->get_block())) {
117  // Check if this symbol is related to the else block.
118  const auto symbol_else =
119  symbol_table->get_block()->find(symbol_pair.first);
120 
121  if (/*symbol_else != symbol_table->get_block()->end() &&*/
122  symbol_pair.second->get_name().compare(
123  symbol_else->second->get_name()) != 0) {
124  // Get name from the previous context.
125  compiler::Symbol* const symbol =
126  ir_context->get_symbol_table()->find_symbol(symbol_pair.first);
127 
128  if (symbol->get_is_pointer()) {
129  throw compiler::fatal_error("PHI MOVE WITH POINTER TYPE.");
130  }
131 
132  if (symbol->get_name()[0] == ir::local_sign[0]) {
133  symbol->set_name(
134  ir::local_sign +
135  std::to_string(
136  ir_context->get_symbol_table()->get_available_id()));
137  }
138 
139  // Set phi blocks.
140  ir_if.emplace_back(ir::op_type::PHI,
141  new ir::Operand(symbol->get_name()),
142  new ir::Operand(symbol_pair.second->get_name()));
143  ir_if.back().set_phi_block(ir_end.begin());
144  ir_else.emplace_back(
145  ir::op_type::PHI, new ir::Operand(symbol->get_name()),
146  new ir::Operand(symbol_else->second->get_name()));
147  ir_else.back().set_phi_block(ir_end.begin());
148  }
149  }
150  }
151 
152  // Append to the ir_list.
153  compiler::insert_with_move(ir_list, ir_if);
154  if (ir_else.empty() == false) {
155  ir_list.emplace_back(
156  ir::op_type::JMP,
157  new ir::Operand(".LBB" + std::to_string(scope_uuid_cur) + "_END_IF"));
158  }
159  ir_list.emplace_back(ir::op_type::LBL,
160  ".LBB" + std::to_string(scope_uuid_cur) + "_ELSE:");
161 
162  compiler::insert_with_move(ir_list, ir_else);
163  ir_list.emplace_back(ir::op_type::JMP,
164  ".LBB" + std::to_string(scope_uuid_cur) + "_END_IF");
165  compiler::insert_with_move(ir_list, ir_end);
166 
167  ir_context->leave_scope();
168  } catch (const std::exception& e) {
169  std::cerr << termcolor::red << termcolor::bold << lineno << ": " << e.what()
170  << termcolor::reset << std::endl;
171  // FIXME: Determine which exception should be caught.
172  }
173 }

175 compiler::ir::Operand* compiler::Item_stmt_assign::eval_runtime_helper(
176  compiler::ir::IRContext* const ir_context) const {
177  try {
178  // Cannot evaluate an array / global symbol.
179  if (identifier->get_ident_type() ==
180  compiler::Item_ident::ident_type::ARRAY) {
181  throw compiler::unsupported_operation(
182  "Error: Cannot evaluate an array type!");
183  }
184  compiler::Symbol* const symbol =
185  ir_context->get_symbol_table()->find_symbol(identifier->get_name());
186  if (symbol->get_name()[0] == ir::global_sign[0]) {
187  throw compiler::unsupported_operation(
188  "Error: Cannot evaluate a global symbol!");
189  } else if (symbol->get_type() == compiler::symbol_type::ARRAY_TYPE) {
190  throw compiler::unsupported_operation(
191  "Error: Cannot evaluate an array type!");
192  }
193 
194  ir::Operand* result = expression->eval_runtime(ir_context);
195  compiler::Symbol_const* symbol_const = new compiler::Symbol_const(
196  result->get_value(), compiler::symbol_type::VAR_TYPE,
197  result->get_value(), false);
198  const std::string symbol_name =
199  ir::local_sign +
200  std::to_string(ir_context->get_symbol_table()->get_available_id());
201  symbol->set_name(symbol_name);
202  ir_context->get_symbol_table()->assign_const(symbol_name, symbol_const);
203 
204  return result;
205  } catch (const std::exception& e) {
206  std::cerr << termcolor::red << termcolor::bold << e.what()
207  << termcolor::reset << std::endl;
208  exit(1);
209  }
210 }

230 void compiler::Item_stmt_assign::generate_ir_helper(
231  compiler::ir::IRContext* const ir_context,
232  std::vector<compiler::ir::IR>& ir_list) const {
233  // We need to check the assigner and the assignee.
234  const Item_ident::ident_type type = identifier->get_ident_type();
235 
236  // Case 1: This identifier is a variable.
237  if (Item_ident::ident_type::VARIABLE == type) {
238  ir::Operand* const rhs = expression->eval_runtime(ir_context, ir_list);
239  compiler::Symbol* const symbol =
240  ir_context->get_symbol_table()->find_symbol(identifier->get_name());
241 
242  if (symbol->get_is_pointer() == true) {
243  throw compiler::unsupported_operation(
244  "Error: Cannot assign an expression to a pointer-like object!");
245  }
246  const std::string name = rhs->get_identifier();
247  if (rhs->get_is_var() == true) {
248  // Local <= Variable. ok. SSA update symbol name.
249  if (name[0] == '%' &&
250  (symbol->get_name()[0] == ir::local_sign[0] ||
251  symbol->get_name().substr(0, 4) == ir::arg_sign) &&
252  symbol->get_name()[0] != ir::global_sign[0]) {
253  if (ir_context->is_loop_context()) {
254  } else {
255  symbol->set_name(name);
256  }
257  }
258  } else if (rhs->get_identifier().front() == '@') /* Global variable. */ {
259  symbol->set_name(rhs->get_identifier());
260  ir_list.emplace_back(ir::op_type::MOV,
261  new ir::Operand(rhs->get_identifier()));
262  } else /* Normal const expression. */ {
263  symbol->set_name(
264  ir::local_sign +
265  std::to_string(ir_context->get_symbol_table()->get_available_id()));
266  ir_list.emplace_back(
267  ir::op_type::MOV,
268  new ir::Operand(
269  ir::local_sign +
270  std::to_string(
271  ir_context->get_symbol_table()->get_available_id())),
272  rhs);
273 
274  if (opt_level > 0) {
275  compiler::Symbol_const* const assign = new compiler::Symbol_const(
276  symbol->get_name(), compiler::symbol_type::VAR_TYPE, "");
277  ir_context->get_symbol_table()->assign_const(identifier->get_name(),
278  assign);
279  }
280  }
281  } else if (Item_ident::ident_type::ARRAY == type) {
282  // Get result.
283  ir::Operand* const res = identifier->eval_runtime(ir_context, ir_list);
284  throw compiler::unimplemented_error("Sorry, this is not yet supported:(");
285  }
286 }

288 void compiler::Item_stmt_while::generate_ir_helper(
289  compiler::ir::IRContext* const ir_context,
290  std::vector<compiler::ir::IR>& ir_list) const {
291  try {
292  // Enter a scope.
293  ir_context->enter_scope();
294  // Create a loop label.
295  const uint32_t scope_id =
296  ir_context->get_symbol_table()->get_top_scope_uuid();
297  ir_context->add_loop_label(std::to_string(scope_id));
298 
299  // Step 1: Copy the previous context.
300  compiler::ir::IRContext* const ir_context_backup =
301  new compiler::ir::IRContext(*ir_context);
302  std::vector<ir::IR> ir_list_before;
303 
304  // Step 2: Create condition.
305  compiler::ir::IRContext* const ir_context_conditional =
306  new compiler::ir::IRContext(*ir_context_backup);
307  std::vector<compiler::ir::IR> ir_conditional;
308  ir_conditional.emplace_back(
309  compiler::ir::op_type::LBL,
310  ".LB" + ir_context->get_top_loop_label() + "LOOP_BEGIN");
311  const compiler::ir::BranchIR branch_ir =
312  condition->eval_cond(ir_context_conditional, ir_conditional);
313 
314  // Step 3: Create jump instruction.
315  std::vector<compiler::ir::IR> ir_jump;
316  ir_jump.emplace_back(ir::op_type::JMP,
317  ".LB" + std::to_string(scope_id) + "LOOP_END");
318 
319  ir_context->leave_scope();
320  } catch (const std::exception& e) {
321  std::cerr << termcolor::red << termcolor::bold << lineno << ": " << e.what()
322  << termcolor::reset << std::endl;
323  }
